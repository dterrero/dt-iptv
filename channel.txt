server {
    listen 80;
    server_name _;

    # Optional: logging
    access_log /opt/iptv/logs/iptv_access.log;
    error_log /opt/iptv/logs/iptv_error.log;

    # Reverse proxy for IPTV channels
    location /streams/ {
        proxy_pass http://SOURCE_IP_OR_DOMAIN/;  # replace with actual source URL
        proxy_buffering off;
        proxy_cache off;

        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }
}
